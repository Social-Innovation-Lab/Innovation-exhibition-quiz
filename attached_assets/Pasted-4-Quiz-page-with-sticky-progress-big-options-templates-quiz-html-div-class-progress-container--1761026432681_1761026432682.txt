static/main.js
(function(){
  const form = document.getElementById('quizForm');
  if(!form) return;
  const qids = [...new Set([...form.querySelectorAll('input[type=radio]')].map(i => i.name))];
  const answered = new Set();
  const ansCount = document.getElementById('ansCount');
  const barFill = document.getElementById('barFill');
  const submitBtn = document.getElementById('submitBtn');

  form.addEventListener('change', (e)=>{
    if(e.target.type === 'radio'){
      answered.add(e.target.name);
      const count = answered.size;
      ansCount.textContent = count;
      barFill.style.width = ((count/qids.length)*100).toFixed(1)+'%';
      submitBtn.disabled = count < qids.length;
    }
  });

  // Prevent accidental back/refresh
  let submitted = false;
  form.addEventListener('submit', ()=>{ submitted = true; submitBtn.disabled = true; submitBtn.textContent='Submitting…'; });
  window.addEventListener('beforeunload', (e)=>{ if(!submitted){ e.preventDefault(); e.returnValue=''; } });

  // Prevent pinch zoom double-tap shenanigans
  document.addEventListener('gesturestart', e => e.preventDefault());
})();

6) Make it a simple PWA (caches assets, works if Wi-Fi hiccups)
Create static/icons/icon-192.png & icon-512.png.
manifest.json
{
  "name": "BRAC Exhibition Quiz",
  "short_name": "BRAC Quiz",
  "start_url": "/",
  "display": "standalone",
  "background_color": "#ffffff",
  "theme_color": "#e31837",
  "icons": [
    { "src": "/static/icons/icon-192.png", "sizes": "192x192", "type": "image/png" },
    { "src": "/static/icons/icon-512.png", "sizes": "512x512", "type": "image/png" }
  ]
}
static/sw.js
const CACHE='brac-quiz-v1';
const ASSETS=[
  '/', '/static/styles.css', '/static/main.js',
  '/manifest.json', '/static/icons/icon-192.png', '/static/icons/icon-512.png'
];
self.addEventListener('install', e=>{
  e.waitUntil(caches.open(CACHE).then(c=>c.addAll(ASSETS)));
});
self.addEventListener('fetch', e=>{
  e.respondWith(
    caches.match(e.request).then(r => r || fetch(e.request))
  );
});
Register the service worker (add at end of templates/index.html and quiz.html):
<script>
if('serviceWorker' in navigator){
  navigator.serviceWorker.register('/static/sw.js').catch(()=>{});
}
</script>
Note: We only cache static assets. Submissions still go to the server so scores/winners are recorded.
7) Concurrency & reliability (multiple tablets at once)
Sessions: Keep server-side session (Flask) and always carry attempt_id in URL for /result.
SQLite safety: Open a new connection per request (check_same_thread=False), enable WAL mode at init:
conn.execute("PRAGMA journal_mode=WAL;")
conn.execute("PRAGMA synchronous=NORMAL;")
Rotation: Use the per-programme rotation queue as described earlier so tablets don’t collide on the same next question; store rotation state in DB, not memory.
